### 1. 消息模型设计

**logic/dao/message.go**：
```go
package dao

import (
	"time"
	"gorm.io/gorm"
)

// Message 消息模型
type Message struct {
	ID        uint      `gorm:"primaryKey" json:"id"`
	Type      int       `json:"type"` // 1: 聊天消息, 2: 私信消息, 3: 系统消息
	Sender    string    `json:"sender"`
	Receiver  string    `json:"receiver"`
	Content   string    `json:"content"`
	Timestamp time.Time `json:"timestamp"`
}

// TableName 设置表名
func (Message) TableName() string {
	return "messages"
}

// CreateMessage 创建消息
func CreateMessage(msg *Message) error {
	return db.Create(msg).Error
}

// GetMessages 获取消息列表
func GetMessages(limit, offset int) ([]Message, error) {
	var messages []Message
	err := db.Order("timestamp desc").Limit(limit).Offset(offset).Find(&messages).Error
	return messages, err
}

// GetPrivateMessages 获取私信消息
func GetPrivateMessages(sender, receiver string, limit, offset int) ([]Message, error) {
	var messages []Message
	err := db.Where("(sender = ? AND receiver = ?) OR (sender = ? AND receiver = ?)", 
		sender, receiver, receiver, sender).
		Order("timestamp desc").
		Limit(limit).
		Offset(offset).
		Find(&messages).Error
	return messages, err
}

// GetMessageCount 获取消息总数
func GetMessageCount() (int64, error) {
	var count int64
	err := db.Model(&Message{}).Count(&count).Error
	return count, err
}

// GetPrivateMessageCount 获取私信消息总数
func GetPrivateMessageCount(sender, receiver string) (int64, error) {
	var count int64
	err := db.Model(&Message{}).
		Where("(sender = ? AND receiver = ?) OR (sender = ? AND receiver = ?)", 
			sender, receiver, receiver, sender).
		Count(&count).Error
	return count, err
}
```

### 2. 数据库表结构

**db/db.go**：
```go
package db

import (
	"gochat/logic/dao"
	"gorm.io/driver/sqlite"
	"gorm.io/gorm"
)

var (
	// db 数据库连接
	db *gorm.DB
)

// Init 初始化数据库
func Init() error {
	var err error
	db, err = gorm.Open(sqlite.Open("gochat.db"), &gorm.Config{})
	if err != nil {
		return err
	}
	
	// 自动迁移
	err = db.AutoMigrate(&dao.User{}, &dao.Message{})
	if err != nil {
		return err
	}
	
	return nil
}

// GetDB 获取数据库连接
func GetDB() *gorm.DB {
	return db
}

// Close 关闭数据库连接
func Close() error {
	sqlDB, err := db.DB()
	if err != nil {
		return err
	}
	return sqlDB.Close()
}
```

### 4. 历史记录查询

**api/handler/message.go**：
```go
package handler

import (
	"gochat/logic"
	"gochat/tools"
	"net/http"
	"strconv"
	"github.com/gin-gonic/gin"
)

// GetMessagesRequest 获取消息请求
type GetMessagesRequest struct {
	Limit  int `form:"limit" binding:"required"`
	Offset int `form:"offset" binding:"required"`
}

// GetPrivateMessagesRequest 获取私信消息请求
type GetPrivateMessagesRequest struct {
	Receiver string `form:"receiver" binding:"required"`
	Limit    int    `form:"limit" binding:"required"`
	Offset   int    `form:"offset" binding:"required"`
}

// GetMessages 获取消息列表
func GetMessages(c *gin.Context) {
	limitStr := c.DefaultQuery("limit", "20")
	offsetStr := c.DefaultQuery("offset", "0")
	
	limit, err := strconv.Atoi(limitStr)
	if err != nil {
		tools.BadRequest(c, "参数错误")
		return
	}
	
	offset, err := strconv.Atoi(offsetStr)
	if err != nil {
		tools.BadRequest(c, "参数错误")
		return
	}
	
	// 获取消息列表
	messages, err := logic.GetMessages(limit, offset)
	if err != nil {
		tools.InternalServerError(c, "获取消息失败")
		return
	}
	
	// 获取消息总数
	count, err := logic.GetMessageCount()
	if err != nil {
		tools.InternalServerError(c, "获取消息总数失败")
		return
	}
	
	tools.Success(c, gin.H{
		"messages": messages,
		"count":    count,
	})
}

// GetPrivateMessages 获取私信消息
func GetPrivateMessages(c *gin.Context) {
	sender := c.Query("sender")
	receiver := c.Query("receiver")
	limitStr := c.DefaultQuery("limit", "20")
	offsetStr := c.DefaultQuery("offset", "0")
	
	if sender == "" || receiver == "" {
		tools.BadRequest(c, "参数错误")
		return
	}
	
	limit, err := strconv.Atoi(limitStr)
	if err != nil {
		tools.BadRequest(c, "参数错误")
		return
	}
	
	offset, err := strconv.Atoi(offsetStr)
	if err != nil {
		tools.BadRequest(c, "参数错误")
		return
	}
	
	// 获取私信消息列表
	messages, err := logic.GetPrivateMessages(sender, receiver, limit, offset)
	if err != nil {
		tools.InternalServerError(c, "获取私信消息失败")
		return
	}
	
	// 获取私信消息总数
	count, err := logic.GetPrivateMessageCount(sender, receiver)
	if err != nil {
		tools.InternalServerError(c, "获取私信消息总数失败")
		return

	tools.Success(c, gin.H{
```

### 5. 更新路由

**api/router/router.go**：
```go
package router

import (
	"gochat/api/handler"
	"github.com/gin-gonic/gin"
)

// SetupRouter 设置路由
func SetupRouter() *gin.Engine {
	r := gin.Default()
	
	// 跨域中间件
	r.Use(func(c *gin.Context) {
		c.Writer.Header().Set("Access-Control-Allow-Origin", "*")
		c.Writer.Header().Set("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS")
		c.Writer.Header().Set("Access-Control-Allow-Headers", "Content-Type, Authorization")
		
		if c.Request.Method == "OPTIONS" {
			c.AbortWithStatus(204)
			return
		}
		
		c.Next()
	})
	
	// API 路由组
	api := r.Group("/api")
	{
		// 用户相关路由
		user := api.Group("/user")
		{
			user.POST("/login", handler.Login)
			user.POST("/register", handler.Register)
		}
		
		// 消息相关路由
		message := api.Group("/message")
		{
			message.POST("/push", handler.PushMessage)
			message.GET("/list", handler.GetMessages)
			message.GET("/private", handler.GetPrivateMessages)
		}
	}
	
	return r
}
```

### 6. 更新逻辑层 RPC 服务

**logic/rpc.go**：
```go
package logic

import (
	"gochat/logic/dao"
	"gochat/proto"
	"gochat/tools"
	"github.com/smallnest/rpcx/server"
)

// LogicService 逻辑服务
type LogicService struct{}

// Login 登录
func (s *LogicService) Login(args *proto.LoginArgs, reply *proto.LoginReply) error {
	// 根据用户名获取用户
	user, err := dao.GetUserByUsername(args.Username)
	if err != nil {
		reply.Error = "用户名或密码错误"
		return nil
	}
	
	// 验证密码
	if user.Password != args.Password {
		reply.Error = "用户名或密码错误"
		return nil
	}
	
	// 生成 token
	reply.Token = args.Username
	reply.Username = user.Username
	return nil
}

// Register 注册
func (s *LogicService) Register(args *proto.RegisterArgs, reply *proto.RegisterReply) error {
	// 检查用户名是否已存在
	_, err := dao.GetUserByUsername(args.Username)
	if err == nil {
		reply.Error = "用户名已存在"
		return nil
	}
	
	// 创建用户
	user, err := dao.CreateUser(args.Username, args.Password)
	if err != nil {
		reply.Error = "注册失败"
		return nil
	}
	
	// 生成 token
	reply.Token = user.Username
	reply.Username = user.Username
	return nil
}

// PushMessage 推送消息
func (s *LogicService) PushMessage(args *proto.PushMessageArgs, reply *proto.PushMessageReply) error {
	// 保存消息
	err := SaveMessage(int(MessageTypeChat), "system", "", args.Message)
	if err != nil {
		reply.Error = "消息保存失败"
		return nil
	}
	
	// 发布消息
	err = PublishMessage(args.Message)
	if err != nil {
		reply.Error = "消息发送失败"
		return nil
	}
	return nil
}

// StartRpcServer 启动 RPC 服务器
func StartRpcServer() error {
	// 创建服务器
	s := server.NewServer()
	
	// 注册服务
	s.RegisterName("LogicRpc", new(LogicService), "")
	
	// 注册到 etcd
	err := tools.RegisterServer(s, "LogicRpc", "127.0.0.1:6900")
	if err != nil {
		return err
	}
	
	// 启动服务器
	return s.Serve("tcp", "127.0.0.1:6900")
}
```
